System Logic


                    #include Wire.h
                    #include LiquidCrystal_I2C.h
                    #include DHT.h
                    
                    // Define Sensor Pins
                    #define DHTPIN 15       // DHT22 connected to GPIO 15
                    #define DHTTYPE DHT22
                    #define PIR_PIN 14      // PIR motion sensor connected to GPIO 14
                    #define SOIL_PIN 32     // Potentiometer (simulating soil moisture) to GPIO 32 (A0)
                    
                    // Initialize DHT sensor
                    DHT dht(DHTPIN, DHTTYPE);
                    
                    // Initialize LCD (I2C address 0x27, 16 columns, 2 rows)
                    LiquidCrystal_I2C lcd(0x27, 16, 2);
                    
                    void setup() {
                        Serial.begin(115200);
                        dht.begin();
                        pinMode(PIR_PIN, INPUT);
                        
                        // Initialize LCD
                        lcd.init();
                        lcd.backlight();
                        lcd.setCursor(0, 0);
                        lcd.print("Initializing...");
                        delay(2000);
                    }
                    
                    void loop() {
                        // Read Temperature & Humidity
                        float temperature = dht.readTemperature();
                        float humidity = dht.readHumidity();
                        
                        // Check if DHT22 is returning valid values
                        if (isnan(temperature) || isnan(humidity)) {
                            Serial.println("DHT22 error reading data!");
                            lcd.clear();
                            lcd.setCursor(0, 0);
                            lcd.print("DHT22 Error!");
                            delay(2000);
                            return;
                        }
                    
                        // Read PIR Sensor
                        int motion = digitalRead(PIR_PIN);
                    
                        // Read Soil Moisture (Potentiometer)
                        int soilMoistureValue = analogRead(SOIL_PIN);
                        int moisturePercent = map(soilMoistureValue, 0, 4095, 100, 0); // Convert to %
                    
                        // Display Readings on Serial Monitor
                        Serial.print("Temp: "); Serial.print(temperature); Serial.println("°C");
                        Serial.print("Humidity: "); Serial.print(humidity); Serial.println("%");
                        Serial.print("Motion: "); Serial.println(motion ? "Yes" : "No");
                        Serial.print("Soil Moisture: "); Serial.print(moisturePercent); Serial.println("%");
                        
                        // Display Readings on LCD
                        lcd.clear();
                        lcd.setCursor(0, 0);
                        lcd.print("Temp: ");
                        lcd.print(temperature);
                        lcd.print("C");
                    
                        lcd.setCursor(0, 1);
                        lcd.print("Soil: ");
                        lcd.print(moisturePercent);
                        lcd.print("% ");
                    
                        delay(2000); // Update every 2 seconds
                    
                        lcd.clear();
                        lcd.setCursor(0, 0);
                        lcd.print("Humidity: ");
                        lcd.print(humidity);
                        lcd.print("%");
                    
                        lcd.setCursor(0, 1);
                        lcd.print("Motion: ");
                        lcd.print(motion ? "YES" : "NO");
                    
                        delay(2000); // Update every 2 seconds
                    }
